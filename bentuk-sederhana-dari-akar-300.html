[tex] \frac{ \sqrt{300} \times 2 \sqrt{12} }{12 \sqrt{3} } \\ = \frac{ \sqrt{100 \times 3} \times 2 \sqrt{4 \times 3} }{12 \sqrt{3} } \\ = \frac{10 \sqrt{3} \times 4 \sqrt{3} }{12 \sqrt{3} } \\ = \frac{120}{12 \sqrt{3} } \\ = \frac{10}{ \sqrt{3} } \\ = \frac{10}{ \sqrt{3} } \times \frac{ \sqrt{3} }{ \sqrt{3} } \\ = \frac{10}{3} \sqrt{3} [/tex]
